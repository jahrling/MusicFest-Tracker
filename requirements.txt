# ── PaddleOCR (local vision backend) ─────────────────────────────────────────
# Only needed when VISION_BACKEND=paddleocr in .env.
#
# Install ONE of the following depending on your hardware:
#
#   CPU (any machine, no GPU required — slower):
#     pip install paddleocr paddlepaddle
#
#   GPU (NVIDIA CUDA — recommended once you have a GPU):
#     pip install paddleocr paddlepaddle-gpu
#
# The paddleocr package is the same in both cases; only the backend differs.
# Do NOT install both paddlepaddle and paddlepaddle-gpu at the same time.
#
# paddleocr>=2.7.0
# paddlepaddle>=2.6.0        # CPU
# paddlepaddle-gpu>=2.6.0    # GPU (uncomment instead of paddlepaddle)
